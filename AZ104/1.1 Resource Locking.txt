# CLI to Manage resource locks throught
# Create a management-level lock on the current Azure subscription that prevents deletions
az account lock create \
  --name "No delete" \        # Friendly name for the lock (any string)
  --lock-type CanNotDelete    # Lock type: blocks DELETE operations; read/update still allowed
# List all subscription-level locks in a readable table
az account lock list \
  --output table              # Format the output as a table (other options: json, yaml, etc.)
# Create a lock on a specific resource group named "Hq" that prevents modifications
az group lock create \
  --lock-type ReadOnly \      # Lock type: allows only read operations; blocks create/update/delete
  -n NoModify \               # Short form of --name: the lock's friendly name
  -g Hq                       # Target resource group (-g is short for --resource-group)
# List all subscription-level locks again, showing both CanNotDelete and ReadOnly locks if applicable

# Create a lock on a specific resource within a resource group to make it read-only
az resource lock create \
  --lock-type ReadOnly \                              # Lock type: restricts modifications; only read operations allowed
  -n NoModify \                                       # Short form of --name: friendly name for the lock
  -g Hq \                                             # Resource group where the resource resides
  --resource storacctyhZ44 \                          # Name of the resource to lock (in this case, a storage account)


# Power Shell Manage resource locks throught 
# Create a ReadOnly lock at the resource group level using PowerShell
New-AzResourceLock `
  -LockName NoModify `                     # Name of the lock
  -LockLevel ReadOnly `                    # Lock level: restricts modifications, allows read-only access
  -ResourceGroupName Hq `                  # Target resource group
  -Force                                   # Suppresses confirmation prompts
# Retrieve all resource locks in the current context
Get-AzResourceLock                          # Lists all locks applied at subscription, resource group, or resource level
# Create a ReadOnly lock at the subscription level using PowerShell
New-AzResourceLock `
  -LockName NoModify `                     # Name of the lock
  -LockLevel ReadOnly `                    # Lock level: read-only access only
  -Scope /subscriptions/00da78ac-9d0e-427b-80da-e1b07c749f72 ` # Full scope path for the subscription
  -Force                                   # Suppresses confirmation prompts
# Remove a lock from a resource group using PowerShell
Remove-AzResourceLock `
  -ResourceGroupName Hq `                  # Resource group where the lock exists
  -Name NoModify `                         # Name of the lock to remove
  -Force                                   # Suppresses confirmation prompts
