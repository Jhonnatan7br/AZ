
# Cost management & Budget setting
# Cost management per subscription (Purchase reservations, AZ pricing calculator)
- Ways to save on monthly charges 
- VM cost quoted per hour, charged by second
- Storage incurred 24/7
- Set budget for monthly spend notifications
- Enable VM auto autodown
- Scale vertical always is more expensive

- Instance pricing: VM, SQL DB, Storage
- Reseved pricing: Commitment of 1-3 years, recommended for constant resource usage, for VMs pricing applies up to purchased quantity
- AZ spot VMs: Unused compute capacity at up to a 90% discount, not guaranteed, could have interruptions, when exceed VM could be deallocated or deleted (Setpolicy) 

- When it is being set resource creating, specify Image, SIze, Spot discount, Region, Term, Billing frecuency (Upfront: year, monthly)

# Tagging from CLI
# View tags
az tag list
az tag list --query [].tagName
# Create tag
az tag create --name NAME
az tag create --name ClientType
# Add value to tag
az tag add-value --name ClientType --value VALUE
az tag add-value --name ClientType --value Govt
# Add tag to resource
az RESOURCE TYPE update -g GROUP -n NAME --set tags.TAGNAME=VALUE
az network vnet update -g hq -n vnet1 --set tags.ClientType=Govt
# Show tags of resource
az RESOURCE TYPE show -g GROUP -n NAME --query tags
az network vnet show -g hq -n vnet --query tags

# Tagging from PowerShell
# ===========================
# Azure PowerShell (Az) - Tagging Cheatsheet
# ===========================
# Prerequisites
# Install and connect (run once per environment/session)
# Install-Module -Name Az -Scope CurrentUser
# Connect-AzAccount
# Select-AzSubscription -SubscriptionId "<your-subscription-id>"
# ---------------------------
# 1) View tag definitions (subscription-level)
# ---------------------------
# List all tag definitions (names and values)
Get-AzTag
# Just the tag names
(Get-AzTag).Properties.Tags | Select-Object -ExpandProperty TagName
# ---------------------------
# 2) Create a tag definition
# ---------------------------
New-AzTag -Name 'NAME'
New-AzTag -Name 'ClientType'
# ---------------------------
# 3) Add a value to a tag definition
# ---------------------------
# Merge adds this value without removing existing ones
Update-AzTag -Name 'ClientType' -Operation Merge -Value 'VALUE'
Update-AzTag -Name 'ClientType' -Operation Merge -Value 'Govt'
# Other operations:
# Update-AzTag -Name 'ClientType' -Operation Replace -Value 'Govt'   # overwrite all values
# Update-AzTag -Name 'ClientType' -Operation Delete -Value 'Govt'     # remove value
# ---------------------------
# 4) Add tag(s) to a resource (generic)
# ---------------------------
# Replace tags entirely with the hashtable provided:
Set-AzResource `
  -ResourceGroupName 'GROUP' `
  -Name 'NAME' `
  -ResourceType 'PROVIDER/TYPE' `
  -Tag @{ TAGNAME = 'VALUE' } `
  -Force
# Example: VNet 'vnet1' in RG 'hq' -> ClientType=Govt
Set-AzResource `
  -ResourceGroupName 'hq' `
  -Name 'vnet1' `
  -ResourceType 'Microsoft.Network/virtualNetworks' `
  -Tag @{ ClientType = 'Govt' } `
  -Force
# Add multiple tags at once
Set-AzResource `
  -ResourceGroupName 'hq' `
  -Name 'vnet1' `
  -ResourceType 'Microsoft.Network/virtualNetworks' `
  -Tag @{ ClientType='Govt'; Owner='NetworkTeam'; Environment='Prod' } `
  -Force
# ---------------------------
# 5) Show tags of a resource (generic)
# ---------------------------
(Get-AzResource `
  -ResourceGroupName 'GROUP' `
  -Name 'NAME' `
  -ResourceType 'PROVIDER/TYPE').Tags
# Example: show tags of VNet 'vnet1' in RG 'hq'
(Get-AzResource `
  -ResourceGroupName 'hq' `
  -Name 'vnet1' `
  -ResourceType 'Microsoft.Network/virtualNetworks').Tags
# ---------------------------
# Optional: Resource-specific approach for VNets
# ---------------------------
# Get, modify, and apply tags on a VNet using network cmdlets
$vnet = Get-AzVirtualNetwork -Name 'vnet1' -ResourceGroupName 'hq'
$vnet.Tag['ClientType'] = 'Govt'
Set-AzVirtualNetwork -VirtualNetwork $vnet
# Verify
(Get-AzVirtualNetwork -Name 'vnet1' -ResourceGroupName 'hq').Tag
# ---------------------------
