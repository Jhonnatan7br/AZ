# PowerShell manage blob 
- Cloud shell creates automatically an storage account
#visualize storage distribution
dir
# List commands related to storage accounts
get-command *storageaccoun*
# Create a new Azure Storage Account
New-AzStorageAccount
# Retrieve an existing Azure Storage Account
Get-AzStorageAccount
# Store storage account context in a variable
$ctx = Get-AzStorageAccount -R hq -Name storacctyhz44
# List properties of $ctx object
$ctx | get-member -type property
# Extract context property from $ctx
$ctx = $ctx.context
# Display updated $ctx
$ctx
# Create a new storage container named "customers" using context
New-AZStorageContainer -Name "customers" -Context $ctx
# Upload a file as a blob to the storage container from Cloudshell container
Set-AZStorageBlobContent -File "./customer_data.docx" -container customers -blob customer_data.docx -context $ctx
# List blobs inside the 'customers' container and show their names
Get-AzStorageContainer -Name customers -Context $ctx | Get-AzStorageBlob | select name
# Command to set Azure Storage Account properties
Set-AzStorageAccount `
    -ResourceGroupName HQ `      # Resource group name
    -Name storacctyhz44 `        # Storage account name
    -AccessTier Cool             # Access tier set to Cool
